ECharts - Java类库使用说明

ECharts - Java源项目地址：https://github.com/abel533/ECharts

ECharts - Java是一个供Java开发使用的ECharts的开发包，主要目的是方便在Java中构造ECharts中可能用到的全部数据结构。

优点：只需在后端进行构建echarts所需的json，PC端和移动端都可以进行直接使用，维护方便，节省开发时间。

1、添加依赖

<dependency>
  <groupId>com.github.abel533</groupId>
  <artifactId>ECharts</artifactId>
  <version>3.0.0</version>
</dependency>

2、使用示例

使用技巧说明：

  1. 前方ECharts官方实例找到你想要的图形，点进去查看，相关的option数据。按照此option数据结构，接口端进行构建相应的json。
  2. 构建完成的option会有很多null的属性值，这些null值会影响图形展示，需要将这些null值的属性去掉，以下使用JSON工具类进行处理。JSON.parse(JSON.toJSONString( option ))

双饼图：

List<Map<String, Object>> result = rootMapper.selectCountEach(params);
List<PieData> pieData = new ArrayList<>();
List<PieData> pie2Data = new ArrayList<>();
for (Map<String, Object> map : result) {
    pieData.add(new PieData(map.get("name").toString(), map.get("videos")));
    pie2Data.add(new PieData(map.get("name").toString(), map.get("views")));
}
Option option = new Option();
option.color("#c23531", "#f7b32e", "#d48265");

String title = "视频中心各大分类数据统计";
Integer count = params.getDateCount() + 1;
title = title.replaceAll("COUNT", count.toString());
log.info(params.getDateUnit() + "," + DATEUNIT.MONTH.getCode());
if (params.getDateUnit().equals(DATEUNIT.MONTH.getCode())) {
    title = title.replace("FORMART", "月");
} else if (params.getDateUnit().equals(DATEUNIT.DAY.getCode())) {
    title = title.replace("FORMART", "天");
} else if (params.getDateUnit().equals(DATEUNIT.YEAR.getCode())) {
    title = title.replace("FORMART", "年");
}
option.title().text(title).subtext("").x("center").show(true);
option.tooltip().trigger(Trigger.item).formatter("{a} <br/>{b} : {c} ({d}%)");
option.legend("上传量", "播放量");
option.legend().left("center").top("bottom");
option.legend().orient(Orient.horizontal);

Pie pie = new Pie("上传量");
pie.setData(pieData);
pie.center("50%", "30%");
pie.radius("40%");
pie.type(SeriesType.pie);
Pie pie2 = new Pie("播放量");
pie2.setData(pie2Data);
pie2.center("50%", "70%");
pie2.radius("40%");
pie.type(SeriesType.pie);
option.series(pie, pie2);
return option;

环形图：

List<Map<String, Object>> result = rootMapper.countTotal();

Option option = new Option();
option.tooltip(new Tooltip().trigger(Trigger.item).formatter("{b}<br> 预约{c}次 ({d}%)"));
option.legend(new Legend().orient(Orient.horizontal).x("center").y("bottom"));

Pie pie = new Pie();
pie.type(SeriesType.pie);
ItemStyle label = new ItemStyle();

WdyNormal wdyNormal = new WdyNormal();
wdyNormal.show(false);
wdyNormal.setPosition("center");
label.normal(wdyNormal);

TextStyle textStyle = new TextStyle();
textStyle.fontSize(30);
textStyle.fontWeight("bold");
WdyEmphasis wdyEmphasis = new WdyEmphasis();
wdyEmphasis.show(true);
wdyEmphasis.setTextStyle(textStyle);
label.emphasis(wdyEmphasis);
pie.label(label);
pie.center("50%", "40%");
pie.radius(new String[] {"50%", "70%"});


for (Map<String, Object> map : result) {
    pie.data().add(new PieData(map.get("name").toString(), map.get("value").toString()));
    option.legend().data().add(map.get("name").toString());
}

option.series(pie);
return option;

双柱形图：

List<String> xList = new ArrayList<>();
List<Integer> barData = new ArrayList<>();
List<Integer> bar2Data = new ArrayList<>();
List<Map<String, Object>> result = rootMapper.selectCountByTotalOfQuestion(params);
List<Map<String, Object>> result2 = rootMapper.selectCountByTotalOfAnswer(params);
Map<String, Object> map;
for (int i = 0; i < result.size(); i++) {
    map = result.get(i);
    xList.add(map.get("x").toString());
    barData.add(NumberUtils.createInteger(map.get("data").toString()));
    map = result2.get(i);
    bar2Data.add(NumberUtils.createInteger(map.get("data").toString()));
}
// option
Option option = new Option();
option.color("#c23531", "#d48265");
// legend
option.legend("问题数", "回答数");
option.legend().left("center").top("bottom");
option.legend().orient(Orient.horizontal);
// title
String title = "近COUNT个FORMART笃学亭问答次数统计";
Integer count = params.getDateCount() + 1;
title = title.replaceAll("COUNT", count.toString());
log.info(params.getDateUnit() + "," + DATEUNIT.MONTH.getCode());
if (params.getDateUnit().equals(DATEUNIT.MONTH.getCode())) {
    title = title.replace("FORMART", "月");
} else if (params.getDateUnit().equals(DATEUNIT.DAY.getCode())) {
    title = title.replace("FORMART", "天");
} else if (params.getDateUnit().equals(DATEUNIT.YEAR.getCode())) {
    title = title.replace("FORMART", "年");
}
TextStyle textStyle = new TextStyle();
textStyle.color("#32323A");
textStyle.fontSize(16);
option.title().text(title).subtext("").subtextStyle(textStyle).textStyle(textStyle).x("center").show();
// yAxis
option.yAxis(new ValueAxis());
// xAxis
CategoryAxis xAxis = new CategoryAxis();
xAxis.boundaryGap(false);
xAxis.setData(xList);
option.xAxis(xAxis);
// tooltip
option.tooltip().trigger(Trigger.axis).axisPointer().type(PointerType.shadow);
// series
Bar bar = new Bar("问题数");
// bar.barWidth(30);
bar.setData(barData);
Bar bar2 = new Bar("回答数");
// bar.barWidth(30);
bar2.setData(bar2Data);
option.series(bar, bar2);

return option;

List<Map<String, Object>> result = rootMapper.selectCountByTotal(params);

xAxis.setData(xList);
xAxis.name("(年/月)").nameGap(15);
option.xAxis(xAxis);
