使用Docker搭建GitLab服务器

运行镜像

docker run -d
--name gitlab  
--link gitlab-postgresql:postgresql 
--link gitlab-redis:redisio 
--publish 10022:22 --publish 80:80 
--env 'GITLAB_PORT=80' --env 'GITLAB_SSH_PORT=10022' 
--env 'GITLAB_SECRETS_DB_KEY_BASE=long-and-random-alpha-numeric-string' 
--env 'GITLAB_SECRETS_SECRET_KEY_BASE=long-and-random-alpha-numeric-string' 
--env 'GITLAB_SECRETS_OTP_KEY_BASE=long-and-random-alpha-numeric-string' 
--env 'GITLAB_HOST=gitlab.vdaoyun.com' 
--env 'SMTP_ENABLED=true' --env 'SMTP_DOMAIN=www.qq.com' 
--env 'SMTP_HOST=smtp.qq.com'  --env 'SMTP_USER=474607030@qq.com' 
--env 'SMTP_PASS=***' 
--volume /usr/gitlab/gitlab:/home/git/data sameersbn/gitlab:10.2.5
