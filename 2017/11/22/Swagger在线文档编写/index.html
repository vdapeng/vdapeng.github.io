Swagger在线Api文档编写

集成配置

第一步：在项目的pom文件添加以下依赖：

 <dependency>
    <groupId>io.springfox</groupId>
    <artifactId>springfox-swagger2</artifactId>
    <version>2.7.0</version>
</dependency>
<dependency>
    <groupId>io.springfox</groupId>
    <artifactId>springfox-swagger-ui</artifactId>
    <version> 2.7.0 </version>
</dependency>

第二步：添加配置文件

新建SwaggerConfiguration文件，内容如下：

@Configuration 
@EnableSwagger2 // 启用swagger在线文档
public class SwaggerConfiguration {

    @Bean
    public Docket createRestApi() {
        return new Docket(DocumentationType.SWAGGER_2)
                .apiInfo(apiInfo())
                .select()
                .apis(RequestHandlerSelectors.basePackage("com.vdaoyun.hsApi.web.controller.common.member"))// 需要生成的文档根目录，根据自己项目进行修改
                .paths(PathSelectors.any())
                .build();
    }

    private ApiInfo apiInfo() {
        return new ApiInfoBuilder()
                .title("微道云RESTFUL API测试页面")
                .description("在这里可以欢快的测试你写的优秀(LESE)代码了")
                .termsOfServiceUrl("www.vdaoyun.com")
                .contact("微小云")
                .version("1.0")
                .build();
    }

}

第三步：访问测试

完成上面两步之后，基本的API在线文档就已经完成了，访问项目地址。看到如下页面说明集成成功。

Swagger在线文档编写

注释说明

@Api：作用于Controller

@ApiIgnore： 忽略注解标注的类或者方法，不添加到API文档中

@ApiOperation： 展示每个API基本信息 

value api名称  
notes 备注说明

@ApiImplicitParam：用于规定接收参数类型、名称、是否必须等信息 

name 对应方法中接收参数名称  
value 备注说明  
required 是否必须 boolean  
paramType 参数类型 body、path、query、header、form中的一种   
    body 使用@RequestBody接收数据 POST有效  
    path 在url中配置{}的参数  
    query 普通查询参数 例如 ?query=q ,jquery ajax中data设置的值也可以，例如 {query:”q”},springMVC中不需要添加注解接收  
    header 使用@RequestHeader接收数据  
    form 笔者未使用，请查看官方API文档  
dataType 数据类型，如果类型名称相同，请指定全路径，例如 dataType = “java.util.Date”，springfox会自动根据类型生成模型

@ApiImplicitParams：包含多个@ApiImplicitParam

@ApiModelProperty：对模型中属性添加说明，只能使用在类中。 

value 参数名称  
required 是否必须 boolean  
hidden 是否隐藏 boolean   
其他信息和上面同名属性作用相同，hidden属性对于集合不能隐藏，目前不知道原因

@ApiParam：对单独某个参数进行说明，使用在类中或者controller方法中都可以。注解中的属性和上面列出的同名属性作用相同

Controller层代码示例：

@Api(value = "简历模块", description = "简历模块相关接口")
public class WcpResumeController {

    @ApiOperation(value = "列表查询")
    @ApiImplicitParams({
        @ApiImplicitParam(name = "pageNum", value = "页码", defaultValue = "1", paramType = "query", dataTypeClass = Integer.class),
        @ApiImplicitParam(name = "pageSize", value = "每页条数", defaultValue = "10", paramType = "query", dataTypeClass = Integer.class),
        @ApiImplicitParam(name = "order", value = "排序字段", defaultValue= "createDate", paramType = "query", dataTypeClass = String.class),
        @ApiImplicitParam(name = "sort", value = "排序方式", defaultValue = "DESC", paramType = "query", dataTypeClass = String.class),
        @ApiImplicitParam(name = "name", value = "名字", paramType = "query", dataTypeClass = String.class),
        @ApiImplicitParam(name = "userId", value = "管理员用户编号",required = true, paramType = "query", dataTypeClass = String.class),
        @ApiImplicitParam(name = "startDate", value = "开始时间", paramType = "query", dataTypeClass = String.class),
        @ApiImplicitParam(name = "finishDate", value = "结束时间", paramType = "query", dataTypeClass = String.class)
    })
    public AjaxJson select(
            @RequestParam(value = "pageNum", defaultValue = "1", required = false) Integer pageNum,
            @RequestParam(value = "pageSize", defaultValue = "10", required = false) Integer pageSize,
            @RequestParam(value = "order", defaultValue = "createDate", required = false) String order,
            @RequestParam(value = "sort", defaultValue = "DESC", required = false) String sort,
            @RequestParam(value = "name", required = false) String name,
            @RequestParam(value = "userId") String userId,
            @RequestParam(value = "startDate", required = false) String startDate,
            @RequestParam(value = "finishDate", required = false) String finishDate
    ) throws Exception {

    }

}

Model层代码示例：

@ApiModel(value = "简历管理员")
@Table(name = "wcp_resume_admin")
public class WcpResumeAdmin implements Serializable {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    @Column(name = "id")
    @ApiModelProperty(name = "id", value = "主键", hidden = true)
    private Integer id;


    /**
     * 微道云企业编号
     */
    @Column(name = "wcp_business_id")
    @ApiModelProperty(name = "wcpBusinessId", value = "微道云企业编号")
    private Integer wcpBusinessId;

}
